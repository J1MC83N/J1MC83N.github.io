using InteractiveUtils, Interact, Mux, Pkg, Cairo, Images
strstatus() = (io = IOBuffer(); Pkg.status(;io); String(take!(io)))
strversioninfo(;verbose::Bool=false) = (io = IOBuffer(); versioninfo(io;verbose); String(take!(io)))
function app(req)
    ctx = CairoContext(CairoRGBSurface(100,100))
    set_source_rgb(ctx,1,1,1)
    @manipulate for ω = 0:0.1:5
        r = 0:0.1:5
        circle(ctx,100rand(),100rand(),5)
        fill(ctx)
        vbox(
            vskip(1em),
            md"$(strversioninfo())",
            vskip(1em),
            # plot(r,sin.(ω.*r)),
            Gray{Float32}.(rand(Float32,5,5)),
            vskip(1em),
            ctx.surface
            )
    end
end
